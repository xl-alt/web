                            // Curl 请求示例
                            Post url https://happyapi.org/v1/chat/completions
                            Headers Authorization: Bearer YOURKEY
                            data：{
                                "messages": [
                                    {
                                        "role": "user",
                                        "content": "你是谁"  // 描述
                                    }
                                ], // messages 必填
                                "stream": true, //  是否流式输出
                                "model": "gpt-4o",  // 选择模型必填
                                "temperature": 0.5, // 设置AI思维扩展度
                                "presence_penalty": 2
                            }
                            // 语音调用方法如下：

                            voice 代表语速 语调
                            input 输入的字符或者文字
                            示例请求：https://fast-horse-44.deno.dev/v1/audio/speech?model=zh-CN-YunxiNeural&voice=rate:0|pitch:0&input=你好
